library(rpart)
library(rpart.plot)
library(caTools)   
library(Metrics)   


datos <- read.csv(file.choose(), stringsAsFactors = FALSE)

formula_modelo <- precio ~ .
muestra <- sample.split(datos$precio, SplitRatio = 0.8)
train  <- subset(datos, muestra)
test   <- subset(datos, !muestra)

modelo_arbol <- rpart(
  formula_modelo,
  data   = train,
  method = "anova",
  control = rpart.control(
    minsplit = 20,
    cp       = 0.001 
  )
)

cat("\n=== Tabla de complejidad (printcp) ===\n")
printcp(modelo_arbol)

cp_opt <- modelo_arbol$cptable[which.min(modelo_arbol$cptable[,"xerror"]), "CP"]
cat("\nMejor cp sugerido:", cp_opt, "\n")

modelo_podado <- prune(modelo_arbol, cp = cp_opt)

rpart.plot(modelo_podado, type = 4, extra = 101)

pred_test <- predict(modelo_podado, newdata = test)

RMSE <- sqrt(mean((test$precio - pred_test)^2))
MAE  <- mean(abs(test$precio - pred_test))

SST  <- sum( (test$precio - mean(test$precio))^2 )
SSE  <- sum( (test$precio - pred_test)^2 )
R2   <- 1 - SSE/SST

cat(sprintf("\n=== MĂ©tricas en TEST ===\nRMSE: %.4f\nMAE: %.4f\nRÂ²: %.4f\n", RMSE, MAE, R2))
